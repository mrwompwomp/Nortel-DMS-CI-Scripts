%Exec to BSY NET links in bulk
COMMAND BSYLNKS(
IF ((LINDEX) = (1)) THEN (
PRINT 'BSYLNKS: BUSY NET LINKS IN BULK';
PRINT 'PARAMETERS: [PLANE][START][LENGTH]';
)ELSE(
IF ((RINDEX) = (3)) THEN (
@1->PL;@2->ST;@3->LEN;
PRINT 'BUSYING LINKS '+(NUMTODECSTR ST)+'-'+(NUMTODECSTR ST+LEN-1);
REPEAT LEN (BSY PL ST;ST+1->ST);
ERASE ST LEN PL;
)ELSE(
PRINT 'INCORRECT PARAMETERS';
)))
