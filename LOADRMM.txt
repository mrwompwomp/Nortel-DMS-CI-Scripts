% LOADRMM - FREDERIC DESAUTELS, 2023
COMMAND LOADRMM(
IF ((LINDEX) = (1)) THEN (
PRINT 'LOADRMM: LOAD THE RMM';
)ELSE(
SEND SINK;
TABLE PMLOADS;
POS RMM10A;
ASSIGN ACTVOL V;
QUIT;
1->A;
DISKUT;
WHILE ((SUBSTR V A 1) = ' ') A+1->A;
LF (STRTOSYM (SUBSTR V A));
SEND PREVIOUS;
ERASE A V;
LOADPM;
))
